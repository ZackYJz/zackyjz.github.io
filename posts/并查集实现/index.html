public class GenericUnionFind<V> {
    //Ê†∑Êú¨ÂíåÂåÖË£ÖÁöÑNodeÂØπË±°ÂØπÂ∫îË°®Ôºà‰ªÖÁî®‰∫éÂàùÂßãÂåñÔºâ
    private Map<V,Node<V>> nodes = new HashMap<V,Node<V>>();

    private static class Node<V>{
        //È´òÂ∫¶ÔºåUnionÊó∂ËøõË°å‰ºòÂåñ
        int rank = 1;
        //ËäÇÁÇπÂ≠òÂÇ®ÁöÑÂÄº
        V value;
        //Áà∂ËäÇÁÇπ
        Node<V> parent = this;
    }

    public void initSet(V v){
        if(nodes.containsKey(v)) {return;}
        nodes.put(v,new Node<>());
    }

    public V find(V v){
        Node<V> node = findNode(v);
        return node == null ? null : node.value;
    }

    private Node<V> findNode(V v){
        Node<V> node = nodes.get(v);
        if( node == null) {return null;}
        while(!Objects.equals(node.value,node.parent.value)){
            node.parent = node.parent.parent;
            node = node.parent;
        }
        return node;
    }

    public void union(V v1,V v2){
        Node<V> p1 = findNode(v1);
        Node<V> p2 = findNode(v2);
        if((p1 == null || p2 == null )||Objects.equals(p1.value,p2.value)) {return;}
        if(p1.rank < p2.rank){
            p1.parent = p2;
        }else if(p2.rank < p1.rank){
            p2.rank = p1.rank;
        }else{
            p1.parent = p2;
            p2.rank++;
        }
    }

    public boolean isSame(V v1,V v2){
        return Objects.equals(find(v1),find(v2));
    }
}
